use serde::{Serialize, Deserialize};

#[derive(Serialize, Deserialize)]
pub struct User {
    id: i64,

    name: String,

    email: String,

    roles: Vec<String>,

    profile: Profile,
}

#[derive(Serialize, Deserialize)]
pub struct Profile {
    age: i64,

    active: bool,
}
